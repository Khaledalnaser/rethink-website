import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const DATA_PATH = path.join(__dirname, '../src/data.json');
const API_KEY = process.env.ANTHROPIC_API_KEY;

// Arabic date formatter
function arabicDate() {
  const d = new Date();
  const months = ['يناير','فبراير','مارس','أبريل','مايو','يونيو','يوليو','أغسطس','سبتمبر','أكتوبر','نوفمبر','ديسمبر'];
  const toArabicNum = n => n.toString().replace(/\d/g, d => '٠١٢٣٤٥٦٧٨٩'[d]);
  return `${toArabicNum(d.getDate())} ${months[d.getMonth()]} ${toArabicNum(d.getFullYear())}`;
}

// Call Claude API
async function callClaude(prompt) {
  const res = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'x-api-key': API_KEY,
      'anthropic-version': '2023-06-01'
    },
    body: JSON.stringify({
      model: 'claude-opus-4-5',
      max_tokens: 1024,
      messages: [{ role: 'user', content: prompt }]
    })
  });
  const data = await res.json();
  if (!res.ok) throw new Error(data.error?.message || 'API error');
  return data.content[0].text.trim();
}

// Generate new article
async function generateArticle() {
  const categories = ['هوية', 'نقد', 'ثقافة', 'فكر', 'مجتمع'];
  const cat = categories[Math.floor(Math.random() * categories.length)];

  const prompt = `أنت كاتب فكري تحليلي لمنصة "أعد التفكير" — منصة عربية تتناول الهوية الثقافية، النقد الاجتماعي، وهندسة الوعي.

اكتب مقالاً جديداً في تصنيف "${cat}" يناسب المنصة.

أجب بـ JSON فقط بهذا الشكل بدون أي نص خارجه:
{
  "title": "عنوان المقال (لا يتجاوز 8 كلمات، جريء ومثير)",
  "desc": "وصف قصير للمقال (جملة واحدة، 15 كلمة كحد أقصى، تشويقي)"
}`;

  const raw = await callClaude(prompt);
  const json = JSON.parse(raw.replace(/```json|```/g, '').trim());
  const reads = ['٥ د', '٦ د', '٧ د', '٨ د', '٩ د', '١٠ د', '١٢ د'];
  return {
    cat,
    title: json.title,
    desc: json.desc,
    date: arabicDate(),
    read: reads[Math.floor(Math.random() * reads.length)]
  };
}

// Main
async function main() {
  console.log('🤖 Rethink Agent starting...');

  if (!API_KEY) throw new Error('ANTHROPIC_API_KEY is not set');

  // Read current data
  const data = JSON.parse(fs.readFileSync(DATA_PATH, 'utf-8'));

  // Generate new content
  console.log('✍️  Generating new article...');
  const article = await generateArticle();
  console.log('✅ Article:', article.title);

  console.log('📰 Generating news item...');
  const news = await generateNews();
  console.log('✅ News:', news.title);

  console.log('💼 Generating business news...');
  const business = await generateBusiness();
  console.log('✅ Business:', business.title);

  // Prepend new content (newest first), keep max 20 items each
  data.articles = [article, ...data.articles].slice(0, 20);
  data.news = [news, ...data.news].slice(0, 20);
  data.business = [business, ...data.business].slice(0, 20);

  // Save updated data
  fs.writeFileSync(DATA_PATH, JSON.stringify(data, null, 2), 'utf-8');
  console.log('💾 data.json updated successfully!');
  console.log('🚀 Done! Vercel will deploy the new content automatically.');
}

main().catch(err => {
  console.error('❌ Agent failed:', err.message);
  process.exit(1);
});
